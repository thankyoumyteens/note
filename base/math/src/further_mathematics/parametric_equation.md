=  \frac{\mathrm{d} (\frac{\psi '(t)}{\varphi '(t)})}{\mathrm{d} x}
=  \frac{\frac{\mathrm{d} (\frac{\psi '(t)}{\varphi '(t)})}{\mathrm{d} t}}{\frac{\mathrm{d} x}{\mathrm{d} t}} \\
\end{align}
