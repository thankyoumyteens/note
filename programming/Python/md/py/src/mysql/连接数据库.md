# 连接数据库

```py
#!/usr/bin/python3
 
import pymysql
 
# 打开数据库连接
db = pymysql.connect(host='127.0.0.1',  # 连接名称, 默认127.0.0.1
                user='root',  # 用户名
                passwd='password',  # 密码
                port=3306,  # 端口, 默认为3306
                db='test',  # 数据库名称
                charset='utf8'  # 字符编码
                )

# 关闭数据库连接
db.close()
```
