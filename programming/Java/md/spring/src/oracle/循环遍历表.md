# 遍历表中所有记录

```sql
-- 任务表主键
DECLARE task_id VARCHAR2(32);
-- 用来暂存任务用户表的每一条记录
task_user_row T_TASK%rowtype;
-- 用来暂存任务用户表的每一个主键
task_user_id_item VARCHAR2(32);

BEGIN

task_id := '1';

for task_user_id_item in (select TASK_USER_ID from T_TASK_USER_REL where TASK_ID=task_id)
loop
    select * into task_user_row from T_TASK_USER where TASK_USER_ID=task_user_id_item.TASK_USER_ID;
    dbms_output.put_line('当前遍历到的用户: ' || task_user_row.USER_NAME);
end loop;

END;
```
