# 遍历表中所有记录

```sql
-- 任务表主键
DECLARE query_time VARCHAR2(32);
-- 用来暂存任务表的每一条记录
task_row T_TASK%rowtype;

BEGIN

query_time := '2020-01-01';

for task_row in (select TASK_USER_ID from T_TASK_INFO where CREATE_TIME > query_time)
loop
    dbms_output.put_line('当前遍历到的任务: ' || task_row.TASK_NAME);
end loop;

END;
```
