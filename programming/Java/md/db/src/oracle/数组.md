# 数组

```sql
declare
    -- 定义数组类型
    TYPE t_varchar32_array IS TABLE OF VARCHAR2(32) INDEX BY BINARY_INTEGER;
    -- 声明数组变量
    task_no_array t_varchar32_array;
    -- 数组长度
    task_no_array_length NUMBER;
    task_row T_TASK_INFO%rowtype;
begin
    for i in 1..10
    loop
        select * into task_row from (select * from T_TASK_INFO order by dbms_random.value) where rownum <= 1;

        -- 数组元素赋值
        task_no_array(i) := task_row.CLUE_ID;
        -- 获取数组长度
        task_no_array_length := task_no_array.COUNT;
        -- 遍历数组元素, 并打印
        FOR j IN 1 .. task_no_array_length LOOP
            dbms_output.put_line(task_no_array(j));
        END LOOP;

    end loop;

    commit;

end;
```
