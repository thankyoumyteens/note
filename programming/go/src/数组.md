# 数组

```go
package main

import "fmt"

func main() {
	// 数组的每个元素都会被初始化为元素类型对应的零值
	var arr0 [3]int
	// 完整写法
	var arr1 [3]int = [3]int{1, 1, 1}
	// 自动确定数组长度
	var arr2 = []int{2, 2, 2}
	// 指定数组长度, 通过索引值进行初始化
	// 没有初始化的元素使用类型默认零值
	// 把索引为1的元素初始化为4
	arr3 := [3]int{1: 3}
	// 不指定数组长度, 通过索引值进行初始化
	// 最大的索引是2, 所以数组长度为3
	arr4 := []int{2: 4}

	fmt.Println(arr0[1], arr0)
	fmt.Println(arr1)
	fmt.Println(arr2)
	fmt.Println(arr3)
	fmt.Println(arr4)
}

// 输出:
// 0 [0 0 0]
// [1 1 1]
// [2 2 2]
// [0 3 0]
// [0 0 4]
```
